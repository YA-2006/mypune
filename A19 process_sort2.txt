// Bubble Sort (Parent)
void bubbleSort(int arr[], int n) {
    int i, j, temp;
    for(i = 0; i < n-1; i++) {
        for(j = 0; j < n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}

// Selection Sort (Child)
void selectionSort(int arr[], int n) {
    int i, j, min, temp;
    for(i = 0; i < n-1; i++) {
        min = i;
        for(j = i+1; j < n; j++) {
            if(arr[j] < arr[min]) {
                min = j;
            }
        }
        temp = arr[i];
        arr[i] = arr[min];
        arr[min] = temp;
    }
}

// Print array
void printArray(int arr[], int n) {
    int i;
    for(i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

int main() {

    int n, i;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    int arr[n], arr_copy[n];

    printf("Enter elements:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
        arr_copy[i] = arr[i]; // copy for child
    }

    pid_t pid = fork();

    if(pid < 0) {
        printf("Fork failed!\n");
        return 1;
    }

    // -------- CHILD PROCESS --------
    else if(pid == 0) {

        printf("\n[Child] PID: %d, Parent PID: %d\n", getpid(), getppid());

        selectionSort(arr_copy, n);

        printf("[Child] Sorted (Selection Sort): ");
        printArray(arr_copy, n);

        // ORPHAN DEMONSTRATION
        sleep(5); // parent exits earlier
        printf("[Child - Orphan] New Parent PID: %d\n", getppid());

        exit(0);
    }

    // -------- PARENT PROCESS --------
    else {

        printf("\n[Parent] PID: %d\n", getpid());

        bubbleSort(arr, n);

        printf("[Parent] Sorted (Bubble Sort): ");
        printArray(arr, n);

        // ZOMBIE DEMONSTRATION
        printf("\n[Parent] Sleeping to create Zombie...\n");
        sleep(5); // child finishes → becomes zombie temporarily

        wait(NULL); // clean zombie
        printf("[Parent] Child collected using wait()\n");
    }

    return 0;
}






















import java.util.Scanner;

public class ProcessSort2 {

    // Bubble Sort (Parent)
    static void bubbleSort(int arr[]) {
        int n = arr.length;
        for(int i = 0; i < n-1; i++) {
            for(int j = 0; j < n-i-1; j++) {
                if(arr[j] > arr[j+1]) {
                    int temp = arr[j];
                    arr[j] = arr[j+1];
                    arr[j+1] = temp;
                }
            }
        }
    }

    // Selection Sort (Child)
    static void selectionSort(int arr[]) {
        int n = arr.length;
        for(int i = 0; i < n-1; i++) {
            int min = i;
            for(int j = i+1; j < n; j++) {
                if(arr[j] < arr[min]) {
                    min = j;
                }
            }
            int temp = arr[i];
            arr[i] = arr[min];
            arr[min] = temp;
        }
    }

    static void printArray(int arr[]) {
        for(int i = 0; i < arr.length; i++)
            System.out.print(arr[i] + " ");
        System.out.println();
    }

    public static void main(String[] args) throws Exception {

        // ---------- CHILD MODE ----------
        if(args.length > 0 && args[0].equals("child")) {

            int n = args.length - 1;
            int arr[] = new int[n];

            for(int i = 0; i < n; i++) {
                arr[i] = Integer.parseInt(args[i+1]);
            }

            System.out.println("[Child] PID: " + ProcessHandle.current().pid());

            selectionSort(arr);

            System.out.print("[Child] Sorted (Selection Sort): ");
            printArray(arr);

            // ORPHAN DEMO
            Thread.sleep(5000);
            System.out.println("[Child] Finished (Orphan simulation)");

            return;
        }

        // ---------- PARENT MODE ----------
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter number of elements: ");
        int n = sc.nextInt();

        int arr[] = new int[n];

        System.out.println("Enter elements:");
        for(int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        System.out.println("\n[Parent] PID: " + ProcessHandle.current().pid());

        // Parent sorting
        bubbleSort(arr);
        System.out.print("[Parent] Sorted (Bubble Sort): ");
        printArray(arr);

        // Prepare child arguments
        String argsForChild[] = new String[n + 3];
        argsForChild[0] = "java";
        argsForChild[1] = "ProcessSort";
        argsForChild[2] = "child";

        for(int i = 0; i < n; i++) {
            argsForChild[i + 3] = String.valueOf(arr[i]);
        }

        // Create child process
        ProcessBuilder pb = new ProcessBuilder(argsForChild);
        pb.inheritIO();

        java.lang.Process p = pb.start();

        // ZOMBIE DEMO (delay wait)
        System.out.println("\n[Parent] Sleeping before wait (Zombie simulation)...");
        Thread.sleep(5000);

        p.waitFor(); // wait for child
        System.out.println("[Parent] Child process finished.");
    }
}
